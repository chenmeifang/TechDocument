/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function(s) {
    var left = 0;
    var right = 0;
    var map = new Map();
    var res = 0;
    while(right < s.length) {
        let str = s[right];
        right++;
        map.set(str, map.get(str) === undefined ? 1 : map.get(str) + 1)
        while(map.get(str) > 1) {
            // 窗口要缩小
            let d = s[left];
            left++;
            map.set(d, map.get(d)-1);
        }
        res = Math.max(res, right - left);
    }
    return res
};
